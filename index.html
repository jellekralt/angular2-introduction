Todo.$inject = [];

/* @ngInject */
function Todo() {

    var directive = {
        scope: {},
        restrict: 'E',
        templateUrl: 'todos/todo.html',
        controller: TodoController,
        controllerAs: 'vm'
    };

    return directive;
}

TodoController.$inject = [];

/* @ngInject */
function TodoController() {
    var vm = this;

    vm.label = '';

    vm.todos = [{
        label: 'Foo',
        complete: false
    }];

    vm.deleteItem = function (index) {
        this.todos.splice(index, 1);
    };
}
                    

app.TodoComponent =
    ng.core.Component({
        selector: 'todo',
        templateUrl: 'todos/todo.html'
    })
    .Class({
        constructor: function () {
            this.label = '';
            this.todos = [{
                label: 'Foo',
                complete: false
            }];
        },
        deleteItem: function (index) {
            this.todos.splice(index, 1);
        }
    });
